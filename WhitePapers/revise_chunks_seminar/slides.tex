\newcommand{\tablearrowright}[4]{%
  \tikz[overlay,remember picture]{
  \draw ($(pic cs:#1) + (5pt,4pt)$) edge[bend right=45,-stealth]
        ($(pic cs:#2) + (5pt,4pt)$)
        node [highlight,right,xshift=5pt,yshift=7pt,#4] {#3};}
}

}
